CREATE TABLE EMPLOYEE (
	FName	INTEGER,
	Minit	INTEGER,
	LName	INTEGER,
	Sex	INTEGER,
	Salary	INTEGER,
	Address	INTEGER,
	Bdate	INTEGER,
	Ssn	INTEGER	NOT NULL,
	FK1_Number	INTEGER,
	FK2_Ssn	INTEGER,
PRIMARY KEY (Ssn) );

CREATE TABLE DEPARTMENT (
	Name	INTEGER	NOT NULL,
	Number	INTEGER	NOT NULL,
	Start_date	INTEGER,
	FK1_Ssn	INTEGER	NOT NULL,
PRIMARY KEY (Number),
UNIQUE(Name) );

CREATE TABLE PROJECT (
	Location	INTEGER,
	Name	INTEGER	NOT NULL,
	Number	INTEGER	NOT NULL,
	FK1_Number	INTEGER	NOT NULL,
PRIMARY KEY (Number),
UNIQUE(Name) );

CREATE TABLE DEPENDENT (
	Name	INTEGER	NOT NULL,
	Sex	INTEGER,
	Birth_Date	INTEGER,
	Relationship	INTEGER,
	FK1_Ssn	INTEGER	NOT NULL,
PRIMARY KEY (Name, FK1_Ssn) );

CREATE TABLE WORKS_ON (
	FK1_Ssn	INTEGER	NOT NULL,
	FK2_Number	INTEGER	NOT NULL,
PRIMARY KEY (FK1_Ssn, FK2_Number) );

CREATE TABLE Locations (
	Locations	INTEGER	NOT NULL,
	FK1_Number	INTEGER	NOT NULL,
PRIMARY KEY (Locations, FK1_Number) );
